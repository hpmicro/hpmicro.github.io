179static inline uint32_t count_set_bits(uint32_t value)
180{
181 if (value == 0) {
182 return 0;
183 }
184 return 1 + count_set_bits(value & (value - 1));
185}

195static inline uint32_t get_first_set_bit_from_lsb(uint32_t value)
196{
197 uint32_t i = 0;
198 if (!value) {
199 return 0xFFFFFFFFUL;
200 }
201 while (value && !(value & 0x1)) {
202 value >>= 1;
203 i++;
204 }
205 return i;
206}
207
216static inline uint32_t get_first_set_bit_from_msb(uint32_t value)
217{
218 uint32_t i = 31;
219 if (!value) {
220 return 0xFFFFFFFFUL;
221 }
222 while (value && !(value & 0x80000000)) {
223 value <<= 1;
224 value &= ~1;
225 i--;
226 }
227 return i;
228}
