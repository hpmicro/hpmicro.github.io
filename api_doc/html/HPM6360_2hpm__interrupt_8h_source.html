659/*
660 * @brief Enter Nested IRQ Handling
661 * @note To simplify the logic, Nested IRQ related registers are stored in the stack as below:
662 * MCAUSE - word offset 16 (not used in the vectored mode)
663 * EPC - word offset 17
664 * MSTATUS = word offset 18
665 * MXSTATUS = word offset 19
666 */
667#define ENTER_NESTED_IRQ_HANDLING_M() {\
668 __asm volatile("csrr t6, mepc \n\
669 sw t6, 17*4(sp) \n\
670 csrr t6, mstatus \n\
671 sw t6, 18*4(sp) \n\
672 csrr t6, %0\n\
673 sw t6, 19*4(sp) \n\
674 " : : "i" CSR_MSTATUS);\
675 SAVE_DSP_CONTEXT(); \
676 __asm volatile("li t6, %0\n" : : "i" (CSR_MSTATUS_MIE_MASK)); \
677 __asm volatile("csrrs t6, mstatus, t6\n"); \
678}
679#define COMPLETE_IRQ_HANDLING_M(irq_num) {\
680 __asm volatile("li t0, 0xe4000000 \n\
681 li t1, 0x200004 \n\
682 add t0, t0, t1 \n\
683 lui t1, 0 \n\
684 slli t2, t1, 0xc \n\
685 add t0, t0, t2 \n");\
686 __asm volatile("li t1, %0" : : "i" (irq_num) :); \
687 __asm volatile("sw t1, 0(t0) \n\
688 fence io, io \n\
689 li t6, 1\n\
690 addi t6, t6, -0x800\n\
691 csrrs t6, mie, t6 \n"); \
692}
693
694#define ENTER_NESTED_IRQ_HANDLING_S() {\
695 __asm volatile("csrr t6, sepc \n\
696 sw t6, 17*4(sp) \n\
697 csrr t6, sstatus \n\
698 sw t6, 18*4(sp) \n\
699 csrr t6, %0\n\
700 sw t6, 19*4(sp) \n\
701 " : : "i" CSR_SSTATUS);\
702 SAVE_DSP_CONTEXT(); \
703 __asm volatile("li t6, %0\n" : : "i" (CSR_SSTATUS_SIE_MASK)); \
704 __asm volatile("csrrs t6, sstatus, t6\n"); \
705}
706#define COMPLETE_IRQ_HANDLING_S(irq_num) {\
707 __asm volatile("li t0, 0xe4000000 \n\
708 li t1, 0x201004 \n\
709 add t0, t0, t1 \n\
710 lui t1, 0 \n\
711 slli t2, t1, 0xc \n\
712 add t0, t0, t2 \n");\
713 __asm volatile("li t1, %0" : : "i" (irq_num) :); \
714 __asm volatile("sw t1, 0(t0) \n\
715 fence io, io \n\
716 li t6, 1\n\
717 addi t6, t6, -0x200\n\
718 csrrs t6, sie, t6 \n"); \
719}

722/*
723 * @brief Exit Nested IRQ Handling
724 * @note To simplify the logic, Nested IRQ related registers are stored in the stack as below:
725 * MCAUSE - word offset 16 (not used in the vectored mode)
726 * EPC - word offset 17
727 * MSTATUS = word offset 18
728 * MXSTATUS = word offset 19
729 */
730#define EXIT_NESTED_IRQ_HANDLING_M() { \
731 __asm volatile("csrrci t6, mstatus, 8 \n\
732 lw t6, 18*4(sp) \n\
733 csrw mstatus, t6 \n\
734 lw t6, 17*4(sp) \n\
735 csrw mepc, t6 \n\
736 lw t6, 19*4(sp) \n\
737 csrw %0, t6 \n" : : "i" CSR_MSTATUS);\
738 RESTORE_DSP_CONTEXT(); \
739}

759/* @brief Nested IRQ entry macro : Save CSRs and enable global irq. */
760#define NESTED_IRQ_ENTER() \
761 SAVE_CSR(CSR_MEPC) \
762 SAVE_CSR(CSR_MSTATUS) \
763 SAVE_MXSTATUS() \
764 SAVE_FCSR() \
765 SAVE_UCODE() \
766 set_csr(CSR_MSTATUS, CSR_MSTATUS_MIE_MASK);
767
768/* @brief Nested IRQ exit macro : Restore CSRs */
769#define NESTED_IRQ_EXIT() \
770 clear_csr(CSR_MSTATUS, CSR_MSTATUS_MIE_MASK); \
771 RESTORE_CSR(CSR_MSTATUS) \
772 RESTORE_CSR(CSR_MEPC) \
773 RESTORE_MXSTATUS() \
774 RESTORE_FCSR() \
775 RESTORE_UCODE()
776
777/*
778 * @brief Nested IRQ exit macro : Restore CSRs
779 * @param[in] irq Target interrupt number
780 */
781#define NESTED_VPLIC_COMPLETE_INTERRUPT(irq) \
782do { \
783 clear_csr(CSR_MIE, CSR_MIP_MEIP_MASK); \
784 __plic_complete_irq(HPM_PLIC_BASE, HPM_PLIC_TARGET_M_MODE, irq); \
785 __asm volatile("fence io, io"); \
786 set_csr(CSR_MIE, CSR_MIP_MEIP_MASK); \
787} while (0)
