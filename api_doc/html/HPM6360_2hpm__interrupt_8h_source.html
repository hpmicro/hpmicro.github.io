746/*
747 * @brief Enter Nested IRQ Handling
748 * @note To simplify the logic, Nested IRQ related registers are stored in the stack as below:
749 * MCAUSE - word offset 16 (not used in the vectored mode)
750 * EPC - word offset 17
751 * MSTATUS = word offset 18
752 * MXSTATUS = word offset 19
753 */
754#define ENTER_NESTED_IRQ_HANDLING_M() { \
755 __asm volatile("\n\
756 csrr s2, mepc \n\
757 csrr s3, mstatus \n");\
758 SAVE_FPU_STATE(); \
759 SAVE_DSP_CONTEXT(); \
760 __asm volatile ("\n\
761 c.li a5, 8\n\
762 csrs mstatus, a5\n"); \
763}

765/*
766 * @brief Complete IRQ Handling
767 */
768#define COMPLETE_IRQ_HANDLING_M(irq_num) { \
769 __asm volatile("\n\
770 lui a5, 0x1\n\
771 addi a5, a5, -2048\n\
772 csrc mie, a5\n"); \
773 __asm volatile("\n\
774 lui a4, 0xe4200\n");\
775 __asm volatile("li a3, %0" : : "i" (irq_num) :); \
776 __asm volatile("sw a3, 4(a4)\n\
777 fence io, io\n"); \
778 __asm volatile("csrs mie, a5"); \
779}
