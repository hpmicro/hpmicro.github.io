16static inline hpm_stat_t enet_rgmii_set_clock_delay(ENET_Type *ptr, uint8_t tx_delay, uint8_t rx_delay)
17{
18 hpm_stat_t stat = status_success;
19
20 if (ptr == HPM_ENET0) {
21 HPM_CONCTL->CTRL0 &= ~(CONCTL_CTRL0_ENET0_TXCLK_DLY_SEL_MASK | CONCTL_CTRL0_ENET0_RXCLK_DLY_SEL_MASK);
22 HPM_CONCTL->CTRL0 |= CONCTL_CTRL0_ENET0_RXCLK_DLY_SEL_SET(tx_delay) | CONCTL_CTRL0_ENET0_RXCLK_DLY_SEL_SET(rx_delay);

33static inline hpm_stat_t enet_rmii_enable_clock(ENET_Type *ptr, bool internal)
34{
35 hpm_stat_t stat = status_success;
36
37 /* use an internal PLL clock as reference clock for rmii mode */
38 if (ptr == HPM_ENET0) {
39 if (internal == true) {
40 /* use a pll clock */
41 HPM_CONCTL->CTRL2 |= CONCTL_CTRL2_ENET0_REFCLK_OE_MASK | CONCTL_CTRL2_ENET0_RMII_TXCLK_SEL_MASK;
42 } else {
43 /* use an external clock as reference clock for rmii mode */
44 HPM_CONCTL->CTRL2 |= CONCTL_CTRL2_ENET0_RMII_TXCLK_SEL_MASK; /* use an external clock */
45 }
46 } else if (ptr == HPM_ENET1) {
47 if (internal == true) {
48 /* use a pll clock */
49 HPM_CONCTL->CTRL3 |= CONCTL_CTRL3_ENET1_REFCLK_OE_MASK | CONCTL_CTRL3_ENET1_RMII_TXCLK_SEL_MASK;
50 } else {
51 /* use an external clock as reference clock for rmii mode */
52 HPM_CONCTL->CTRL3 |= CONCTL_CTRL3_ENET1_RMII_TXCLK_SEL_MASK; /* use an external clock */
53 }
54 } else {
55 return status_invalid_argument;
56 }
57
58 return stat;
59}
