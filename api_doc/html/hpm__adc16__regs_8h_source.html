164/* Bitfield definition for register array: BUS_RESULT */
165/*
166 * VALID (RO)
167 *
168 * set after conversion finished if wait_dis is set, cleared after software read.
169 * The first time read with 0 will trigger one new conversion.
170 * If SW read other channel when one channel conversion is in progress, it will not trigger new conversion at other channel, and will get old result with valid 0, also with read_cflct interrupt status bit set.
171 * the result may not realtime if software read once and wait long time to read again
172 */
173#define ADC16_BUS_RESULT_VALID_MASK (0x10000UL)
174#define ADC16_BUS_RESULT_VALID_SHIFT (16U)
175#define ADC16_BUS_RESULT_VALID_GET(x) (((uint32_t)(x) & ADC16_BUS_RESULT_VALID_MASK) >> ADC16_BUS_RESULT_VALID_SHIFT)

177/*
178 * CHAN_RESULT (RO)
179 *
180 * read this register will trigger one adc conversion.
181 * If wait_dis bit is set, SW will get the latest conversion result(not current one) with valid bit is 0, SW need polling valid bit till it's set to get current result
182 * If wait_dis bit is 0, SW can get the current conversion result with holding the bus, valid bit is always set at this mode. this is not recommended if channel sample time is too long
183 */
184#define ADC16_BUS_RESULT_CHAN_RESULT_MASK (0xFFFFU)
185#define ADC16_BUS_RESULT_CHAN_RESULT_SHIFT (0U)
