193static inline void pwm_issue_shadow_register_lock_event(PWM_Type *pwm_x)
194{
195 if (pwm_x->SHCR & PWM_SHCR_SHLKEN_MASK) {
196 /*
197 * if lock shadow register has been enabled in SHCR, it has to set
198 * the lock bit twice to issue shadow register lock event.
199 */
200 pwm_x->SHLK = PWM_SHLK_SHLK_MASK;
201 }
202 pwm_x->SHLK = PWM_SHLK_SHLK_MASK;
203}

213 pwm_x->SHLK = PWM_SHLK_SHLK_MASK;
214}

375static inline void pwm_config_load_counter_shadow_register_trigger(PWM_Type *pwm_x,
376 pwm_shadow_register_update_trigger_t trigger,
377 uint8_t target_cmp_index)
378{
379 if (trigger == pwm_shadow_register_update_on_hw_event) {
380 pwm_x->SHCR = ((pwm_x->SHCR & ~(PWM_SHCR_CNTSHDWSEL_MASK
381 | PWM_SHCR_CNTSHDWUPT_MASK))
382 | PWM_SHCR_CNTSHDWSEL_SET(target_cmp_index)
383 | PWM_SHCR_CNTSHDWUPT_SET(trigger));
384 } else {
385 pwm_x->SHCR = ((pwm_x->SHCR & ~(PWM_SHCR_CNTSHDWUPT_MASK))
386 | PWM_SHCR_CNTSHDWUPT_SET(trigger));
387 }
388}

465static inline void pwm_config_cmp(PWM_Type *pwm_x, uint8_t index, pwm_cmp_config_t *config)
466{
467 pwm_shadow_register_unlock(pwm_x);
468 if (config->mode == pwm_cmp_mode_output_compare) {
469 pwm_x->CMPCFG[index] = PWM_CMPCFG_XCNTCMPEN_SET(config->enable_ex_cmp)
470 | PWM_CMPCFG_CMPSHDWUPT_SET(config->update_trigger);
471 pwm_x->CMP[index] = PWM_CMP_CMP_SET(config->cmp)
472 | PWM_CMP_XCMP_SET(config->ex_cmp)
473 | PWM_CMP_CMPHLF_SET(config->half_clock_cmp)
474 | PWM_CMP_CMPJIT_SET(config->jitter_cmp);
475 } else {
476 pwm_x->CMPCFG[index] |= PWM_CMPCFG_CMPMODE_MASK;
477 }
478}

500static inline void pwm_config_fault_source(PWM_Type *pwm_x, pwm_fault_source_config_t *config)
501{
502 pwm_x->GCR = (pwm_x->GCR & ~(PWM_GCR_FAULTI0EN_MASK | PWM_GCR_FAULTI1EN_MASK
503 | PWM_GCR_FAULTI2EN_MASK | PWM_GCR_FAULTI3EN_MASK
504 | PWM_GCR_FAULTE0EN_MASK | PWM_GCR_FAULTE1EN_MASK
505 | PWM_GCR_FAULTRECEDG_MASK | PWM_GCR_FAULTEXPOL_MASK
506 | PWM_GCR_FAULTRECHWSEL_MASK))
507 | config->source_mask
508 | PWM_GCR_FAULTEXPOL_SET(config->external_fault_active_low)
509 | PWM_GCR_FAULTRECEDG_SET(config->fault_recover_at_rising_edge)
510 | PWM_GCR_FAULTRECHWSEL_SET(config->fault_output_recovery_trigger);
511}
