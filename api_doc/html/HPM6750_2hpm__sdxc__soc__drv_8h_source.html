22static inline void sdxc_wait_card_active(SDXC_Type *base)
23{
24 base->SYS_CTRL |= SDXC_SYS_CTRL_SD_CLK_EN_MASK;
25
26 while (!IS_HPM_BITMASK_SET(base->SYS_CTRL, SDXC_SYS_CTRL_SD_CLK_EN_MASK)) {}
27 /*
28 * Assume that the BUS clock is 200MHz, while in the identification state, the SD clock is 400KHz.
29 * - Reading the register costs about 5ns in best case, hence the loop should be (74 / 400KHz) * 1000000000 / 5
30 * - For robustness, use 50000 here
31 */
32 for (volatile uint32_t i = 0; i < 50000U; i++) {
33 (void) base->CAPABILITIES1;
34 }
35}
