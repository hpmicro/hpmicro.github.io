461/*
462 * @brief Enter Nested IRQ Handling
463 * @note To simplify the logic, Nested IRQ related registers are stored in the stack as below:
464 * MCAUSE - word offset 16 (not used in the vectored mode)
465 * EPC - word offset 17
466 * MSTATUS = word offset 18
467 * MXSTATUS = word offset 19
468 */
469#define ENTER_NESTED_IRQ_HANDLING_M() {\
470 __asm volatile("csrr t6, mepc \n\
471 sw t6, 17*4(sp) \n\
472 csrr t6, mstatus \n\
473 sw t6, 18*4(sp) \n\
474 csrr t6, %0\n\
475 sw t6, 19*4(sp) \n\
476 " : : "i" CSR_MSTATUS);\
477 SAVE_DSP_CONTEXT(); \
478 __asm volatile("li t6, %0\n" : : "i" (CSR_MSTATUS_MIE_MASK)); \
479 __asm volatile("csrrs t6, mstatus, t6\n"); \
480}
481#define COMPLETE_IRQ_HANDLING_M(irq_num) {\
482 __asm volatile("li t0, 0xe4000000 \n\
483 li t1, 0x200004 \n\
484 add t0, t0, t1 \n\
485 lui t1, 0 \n\
486 slli t2, t1, 0xc \n\
487 add t0, t0, t2 \n");\
488 __asm volatile("li t1, %0" : : "i" (irq_num) :); \
489 __asm volatile("sw t1, 0(t0) \n\
490 fence io, io \n\
491 li t6, 1\n\
492 addi t6, t6, -0x800\n\
493 csrrs t6, mie, t6 \n"); \
494}

496/*
497 * @brief Exit Nested IRQ Handling
498 * @note To simplify the logic, Nested IRQ related registers are stored in the stack as below:
499 * MCAUSE - word offset 16 (not used in the vectored mode)
500 * EPC - word offset 17
501 * MSTATUS = word offset 18
502 * MXSTATUS = word offset 19
503 */
504#define EXIT_NESTED_IRQ_HANDLING_M() { \
505 __asm volatile("csrrci t6, mstatus, 8 \n\
506 lw t6, 18*4(sp) \n\
507 csrw mstatus, t6 \n\
508 lw t6, 17*4(sp) \n\
509 csrw mepc, t6 \n\
510 lw t6, 19*4(sp) \n\
511 csrw %0, t6 \n" : : "i" CSR_MSTATUS);\
512 RESTORE_DSP_CONTEXT(); \
513}
