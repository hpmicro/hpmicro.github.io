80static inline void rtc_enable_alarm_interrupt(RTC_Type *base, uint32_t index, bool enable)
81{
82 if (index > 1) {
83 return;
84 }
85
86 uint32_t mask = (index == 0U) ? RTC_ALARM_EN_ENABLE0_MASK : RTC_ALARM_EN_ENABLE1_MASK;
87
88 if (enable) {

91 else {
