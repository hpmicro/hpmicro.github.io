157static inline void trgm_input_filter_config(TRGM_Type *ptr, uint8_t input, trgm_input_filter_t *filter)
158{
159 ptr->FILTCFG[input] = TRGM_FILTCFG_OUTINV_SET(filter->invert)
160 | TRGM_FILTCFG_MODE_SET(filter->mode)
161 | TRGM_FILTCFG_SYNCEN_SET(filter->sync)
162 | TRGM_FILTCFG_FILTLEN_SET(filter->filter_length);
163}

172static inline void trgm_output_update_source(TRGM_Type *ptr, uint8_t output, uint8_t source)
173{
174 ptr->TRGOCFG[output] = (ptr->TRGOCFG[output] & ~TRGM_TRGOCFG_TRIGOSEL_MASK)
175 | TRGM_TRGOCFG_TRIGOSEL_SET(source);
176}

185static inline void trgm_output_config(TRGM_Type *ptr, uint8_t output, trgm_output_t *config)
186{
187 ptr->TRGOCFG[output] = TRGM_TRGOCFG_TRIGOSEL_SET(config->input)
188 | TRGM_TRGOCFG_FEDG2PEN_SET(config->type)
189 | TRGM_TRGOCFG_REDG2PEN_SET(config->type);
190}
