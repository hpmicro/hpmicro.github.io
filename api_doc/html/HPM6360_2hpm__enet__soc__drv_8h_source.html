23static inline hpm_stat_t enet_rmii_enable_clock(ENET_Type *ptr, bool internal)
24{
25 hpm_stat_t stat = status_success;
26
27 /* use an internal PLL clock as reference clock for rmii mode */
28 if (ptr == HPM_ENET0) {
29 if (internal == true) {
30 /* use a pll clock */
31 ptr->CTRL2 |= ENET_CTRL2_ENET0_REFCLK_OE_MASK | ENET_CTRL2_ENET0_RMII_TXCLK_SEL_MASK;
32 } else {
33 /* use an external clock as reference clock for rmii mode */
34 ptr->CTRL2 |= ENET_CTRL2_ENET0_RMII_TXCLK_SEL_MASK; /* use an external clock */
35 }
36 } else {
37 return status_invalid_argument;
38 }
39
40 return stat;
41}
